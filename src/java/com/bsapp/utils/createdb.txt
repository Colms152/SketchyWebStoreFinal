DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10),
"EIRCODE" VARCHAR(100)
);

DROP TABLE "PRODUCT";

CREATE TABLE "PRODUCT"
(
"PRODUCT_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"NAME" VARCHAR(50),
"TYPE" VARCHAR(10),
"LONG_DESCRIPTION" VARCHAR(100),
"CODE" VARCHAR(50),
"STOCK" INT
); 

DROP TABLE "TORDER";

CREATE TABLE "TORDER"
(
"RECORD_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
"ORDER_ID" INT ,   
"CUSTOMER_ID" INT,
"SHIPPING_ID" INT,
"PRODUCT_ID" INT,
"QUANTITY" INT,
"EXTERNELPAY" INT
);
